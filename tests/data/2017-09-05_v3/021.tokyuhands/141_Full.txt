<script type="text/javascript">
$(function(){
	var make_autocomplete = function(term_input) {
		var org_val = term_input.val();
		var autocomplete = term_input.autocomplete({ 
			source: '/suggest',
			delay: 100
		}).data('ui-autocomplete')

		term_input.on("autocompleteselect", function(event, ui) {
			if (ui.item.link) {
				term_input.val(org_val);
				location.href = ui.item.link;
				return false;
			} else {
				term_input.val( ui.item.value );
				term_input.parents('form').submit();
			}
		});

		term_input.on("autocompletefocus", function(event, ui) {
			if ( ! event.keyCode) {
				return;
			}
			if (ui.item.value) {
				if ( autocomplete.xhr ) {
					autocomplete.xhr.abort();
				}
				autocomplete.xhr = $.ajax({
					url:  '/suggest',
					data: {term: ui.item.value, items_only: 1},
					dataType: "json",
					cache: true,
					success: function( data ) {
						autocomplete.menu.element.find('.item-suggest').remove();
						autocomplete._renderMenu(autocomplete.menu.element, data);
						autocomplete.menu.refresh();
					},
					error: function() {
						
					}
				});
			} else {
				term_input.val(autocomplete.term);
				return false;
			}
		});

		autocomplete._renderItem = function(ul, item) {
			$(ul).addClass('suggests');
			if (item.image) {
				return $('<li class="item-suggest">')
					.append('<a href="'+ item.link + '">' + item.image + item.title + '</a>')
					.appendTo(ul);
			} else {
				return $('<li class="keyword-suggest">')
					.append('<a style="border:0;">' + item.value + '</a>')
					.appendTo(ul);
			}
		};
	}

	$('#header_q,#form_q').each(function(index, element) {
		make_autocomplete($(element));
	});
});
</script>
<script type="text/javascript">
$.fn.selects = function(){

	this.each(function(){
		var selects = $(this);
		$('dt',selects).click(function(){
			$('dd',selects).toggle();
		});
		$('dd li',selects).click(function(){
			var text = $(this).html();
			$('dt span',selects).html(text.replace(/^(　| |&nbsp;)+/, ''));
			var category = $(this).data('category');
			$('input', selects).val(category);
			$('dd',selects).hide();

		});
		$(document).bind('click', function(e){
			var $clicked = $(e.target);
			if(! $clicked.parents().hasClass('selects'))
			$('dd',selects).hide();
		});
	});
	return this;
};

$(function(){
	$('.selects').selects();
});
</script>

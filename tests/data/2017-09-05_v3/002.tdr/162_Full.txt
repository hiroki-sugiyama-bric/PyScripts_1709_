<script type="text/javascript">
	/**
	 * エラーメッセージをダイアログ表示する
	 */
	function errorMsgAlertDialog(errorMsg, callback) {
		
		if ( $('.notShowErrorMsgPopup').length > 0 ) {
			return false;
		}

		//エラーメッセージを表示する
		var $errorbox = $('#apiMsg_modalDialog').find('.errorMsgBox');

		//clean the errors
		$errorbox.contents().remove();

		$errorbox.append(errorMsg);

		$.tdrec.dialogSmall({
			useCloseButton : true,
			useBottomCloseButton: true,
			feedInSpeed : 100,
			contents : $('#apiMsg_modalDialog').html()
		}).open();

		$('body').on('click', '.dialogClose', function( e ) {
			$('#modalOverlay, #modalDialog').remove();
			
			if ((typeof callback != "undefined") && $.isFunction(callback)) {
				callback();
			}
		});
	}
</script>

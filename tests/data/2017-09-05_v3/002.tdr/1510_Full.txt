        function setSearch(targetID){

            var $form = $('#' + targetID + '_form');
            var $header = $form.find('.header');
            var $orderSelect = $header.find('.select');
            var $order = $orderSelect.find('select');

            var onlyMyDisney = '';

            if (hash != '') {
                var selectors = hash.split('.');
                if (selectors.length == 5) {
                    categoryVal = selectors[1];
                    onlyMyDisney = selectors[2];
                    sort = selectors[3];
                } else {
                    categoryVal = 'all';
                    onlyMyDisney = 'all';
                    sort = 'recommend';
                }

                var lastSplitPathName = splitPathName[splitPathName.length - 1];

                    onlyMyDisney = selectors[2];
                    sort = selectors[3];

                            }
                        });

        // 縮小率計算
        function list_filter_mydisney_keepGrid(){
            // コンポーネント縮小
            // --------------------------------------------------
            var UA = window.navigator.userAgent.toLowerCase();
            var APPVER = window.navigator.appVersion.toLowerCase();
            var FLG_IE8 = disney.fn.checkIE8();
            var FLG_TRANSITION = disney.fn.checkTransition();
            var W_DEVICE = disney.fn.widthDevice();
            var UA_DEVICE = disney.fn.uaDevice();
            var $BODY = $('body');
            var $STATUS = $('#myStatus');
            var $cmp = $('.setScale');
            var $cmpTab = $('.setScaleTab');
            var $cmpInner = $cmp.find('>div');
            var $cmpInnerTab = $cmpTab.find('>div');
            var wBase = 1200;
            var wBaseTab = 760;
            var wMargin = W_PC - wBase;
            var wMarginTab = W_TAB - wBaseTab;
            
            wBrowser = $(window).width() - wMargin;
            wBrowserTab = $(window).width() - wMarginTab;
            
            if(W_TAB < window.innerWidth){
                if(wBase <= wBrowser){
                    wBrowser = wBase;
                }
                zoomW = wBrowser / wBase;
                $cmpInner.css({
                    webkitTransform: 'scale(' + zoomW + ')',
                    transform: 'scale(' + zoomW + ')'
                }).attr('data-scale', zoomW);
                
                // scaleでの余白切り取り
                $cmp.each(function(e){
                    var cmpH = $(this).find('>div').innerHeight();
                    $(this).css({
                        height: cmpH * zoomW + 3
                    });
                });
                
            }else if(window.innerWidth <= W_SP){
                $cmpTab.css({
                    height: 'auto'
                });
                $cmpTab.css({
                    webkitTransform: 'scale(' + 1 + ')',
                    transform: 'scale(' + 1 + ')'
                }).attr('data-scale', 1);

$(document).ready(function () {

    var STRENGTH_CLASS_NAME_LOW    = 'low';
    var STRENGTH_CLASS_NAME_MIDDLE = 'middle';
    var STRENGTH_CLASS_NAME_HIGH   = 'high';

    updateStrength();
    $("#user_password").keyup(function(key) {
        updateStrength();
    });

    function updateStrength(){
        var pwd = $("#user_password").val(),
            mail_address = $('#mail_address').text() || $('#target_user_mail').val() || $('#mail').val(),
            callback = function(strength) {
                updateUI(strength);
            },
            parameters = {
                password:pwd,
                mailAddress:mail_address,
                enableDictionaryChecker:$("#user_password"),
            };
        passwordStrength.calculate(parameters, callback);

        function updateUI(strength) {
            validator_meter.removeClass(STRENGTH_CLASS_NAME_LOW).removeClass(STRENGTH_CLASS_NAME_MIDDLE).removeClass(STRENGTH_CLASS_NAME_HIGH);
            validator_meter_label.removeClass(STRENGTH_CLASS_NAME_LOW).removeClass(STRENGTH_CLASS_NAME_MIDDLE).removeClass(STRENGTH_CLASS_NAME_HIGH).removeClass(STRENGTH_CLASS_NAME_SHORT);

            if (strength.password.length === 0) {
                validator_meter_label.text("");
                return;
            } else if (strength.password.length < 6) {
                validator_meter_label.addClass(STRENGTH_CLASS_NAME_SHORT);
                validator_meter_label.text("短すぎます");
                return;
            }

            var strengthClasName;
            var strengthText;
            if (strength.level === 1) {
                strengthClasName = STRENGTH_CLASS_NAME_LOW;
                strengthText = "低";
            } else if (strength.level === 2) {
                strengthClasName = STRENGTH_CLASS_NAME_MIDDLE;
                strengthText = "中";
            } else if(strength.level === 3) {
                strengthClasName = STRENGTH_CLASS_NAME_HIGH;
                strengthText = "高";
            } else {
                strengthText = "エラー";
            }

            validator_meter.addClass(strengthClasName);
            validator_meter_label.addClass(strengthClasName);
            validator_meter_label.text(strengthText);
        }
    }

});

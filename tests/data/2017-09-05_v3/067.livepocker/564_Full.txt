function show_pickup_events(data) {
    if (data.result.total_count < 1){
        $('#pickup-event-section').hide();
        return;
    }

    $('#pickup-event-section').show();

    var data_list = data['result']['data_list'];
    var all = [];
    var block = {};
    block.events = [];
    if(data.result.total_count > limit_pu) {
        var total_count = limit_pu;
    }else {
        var total_count = data.result.total_count;
    }

    for (var i = 0; i < total_count; i++) {
        if (i % 2 == 0 && i > 0) {
            all.push(block);
            block = {};
            block.events = [];
        }
        
        block.events.push(data_list[i]);
        if (total_count - 1 == i){//最後の場合
           all.push(block);
        }
    }
    
    // jQuery template.
    var pu_list = $('#eventListFullImgTemplate').tmpl(all).appendTo('#pickup_events_sp');
    if (all.length > 1){
        //exec_slide_load();
    }else{
        //$("ul.pickup-event-list").attr("style", "width: 1000px");
    }

    if (!is_sp) {
        var all_pc = [];
        var block_pc = {};
        block_pc.events = [];
        for (var i = 0; i < total_count; i++) {
            if (i % 4 == 0 && i > 0) {
                all_pc.push(block_pc);
                block_pc = {};
                block_pc.events = [];
            }
            
            block_pc.events.push(data_list[i]);
            if (total_count - 1 == i){//最後の場合
               all_pc.push(block_pc);
            }
        }

        // jQuery template.
        var pu_list_pc = $('#eventListFullImgTemplate').tmpl(all_pc).appendTo('#pickup_events');
        if (all_pc.length > 1){
            //exec_slide_load_pc();
        }else{
            $("#pickup_events .pickup-event-list").addClass('wide');
        }
    } else {
        $('#js_pickup_pc').hide();
    }

}

function common_count() {
    // キャッシュ避けのタイムスタンプ
    var timestamp = new Date().getTime();
    API.get('common/lists?dev=sp&mytimestamp='+timestamp, callback_common_count);
}

function callback_common_count(data) {
    if (data && data['success']) {
        result = data['result'];
        if (data['result']['myticket_count']['count']) {
            if(parseInt(data['result']['myticket_count']['count']) > 0){
                $('.js-myticket-count').show();
                $('.js-myticket-count').text(data['result']['myticket_count']['count']);
            }else{
                $('.js-myticket-count').hide();
            }
        } else {
            $('.js-myticket-count').hide();
        }
        if (data['result']['bookmark_count']['count']) {
            if(parseInt(data['result']['bookmark_count']['count']) > 0){
                $('.js-bookmark-count').show();
                $('.js-bookmark-count').text(data['result']['bookmark_count']['count']);
            }else{
                $('.js-bookmark-count').hide();
            }
        } else {
            $('.js-bookmark-count').hide();
        }

        total = data['result']['today_event']['total_count'];
        if (total < 1) {
            $('#todays_ticket_section').hide();
        }

        //data_list = data['result']['today_event']['data_list'];
        //$('#todayEventTemplate').tmpl(data_list).appendTo('#todayeventlist').trigger("create");

        var count = data['result']['unread_count'];
        if (count > 0) {
            $('.js-notification-count').show();
            $('.js-notification-count').text(count);
        } else {
            $('.js-notification-count').hide();
        }

    }else{
        $('#todays_ticket_section').hide();
    }
}

function update_notice_unread_count() {
    var timestamp = new Date().getTime();
    API.get('notifications/unread_count?mytimestamp='+timestamp, callback_unread_notice_count);
}

function callback_unread_notice_count(data) {
    count = data['result']['unread_count'];
    if (count > 0) {
        $('.js-notification-count').show();
        $('.js-notification-count').text(count);
    } else {
        $('.js-notification-count').hide();
    }
}
